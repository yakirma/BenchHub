    Reads the repo's Croissant/parquet schema and infers field kinds. If it isn't tabular (paired files, packed .npz/.h5) we hand off to the file-tree mapper automatically — or switch to it from the preview. Caches a downscaled preview of the full split (quota-bounded). Browse trending →

    Stage samples in Python, upload in one call. Grab a token from the tokens page. 

    import benchhub as bh
import numpy as np

client  = bh.Client()
creator = client.create_dataset("my-tiny-dataset")

creator.add_field("image", bh.Image, role="input")
creator.add_field("label", bh.Label, role="gt",
                  params={"vocab": ["cat", "dog"]})

for sid, img, lbl in iter_my_data():
    creator.add_sample(
        sid,
        image = bh.Image(img),
        label = bh.Label(lbl),
    )

result = creator.create()
print(result["dataset_id"])
