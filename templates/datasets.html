Build from Python

Stage samples in Python, then upload in one call. Grab your token from the tokens page. 

import benchhub as bh
import numpy as np

client  = bh.Client()                  # env: BENCHHUB_API_TOKEN + BENCHHUB_BASE_URL
creator = client.create_dataset("my-tiny-dataset")

creator.add_field("image",    bh.Image, role="input")
creator.add_field("label",    bh.Label, role="gt",
                  params={"vocab": ["cat", "dog"]})

for sid, img, lbl in iter_my_data():   # your generator
    creator.add_sample(
        sid,
        image = bh.Image(img),         # uint8 (H,W,3)
        label = bh.Label(lbl),
    )

result = creator.create()              # POSTs the ZIP, returns the summary
print(result["dataset_id"])
