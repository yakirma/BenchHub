{# HF-style class selector for label fields. Both render links to the existing label filter; the comparison page intercepts those clicks for in-place AJAX. {{ facet.field }} {{ facet.classes|length }} classes · {{ '{:,}'.format(facet.total) }} {% if not _all_active %}show all{% endif %}

{% else %} {# ---- pills (few classes) ---- #}

{% endif %} {% endfor %} {% endmacro %}
